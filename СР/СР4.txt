//СР4(2) Одноклеточная амеба каждые 3 часа делится на 2 клетки. Определить, сколько амеб будет через 3, 6, 9, 12, …,  24 часа.
#include <iostream>
#include <math.h>
using namespace std;
int main() 
{
    //начальное значение
    int am=1;
    //цикл от 0 до 24 часов
    for(int h=0; h<=24; h+=3)
    {
        //кол-во делений
        int del=h/3;
        //кол-во амеб после деления
        int ams=am*(pow(2, del));
    
    cout<<"\nЧерез "<< h << " часов: "<< ams << " амеб";
    }
    return 0;
}
